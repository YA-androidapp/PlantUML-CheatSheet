repeat
    #CCFFFF:Sign in;
    :Check credential;

    backward :Wrong password;
    note right : username and / or password is incorrect
repeat while (Valid?) is (no) not (yes)

fork
    :Search product;
    :Add to cart;
    split
        :Order;
        :Payment;

        if (Paid?) then (yes)
            :Notify;
        else
            :Payment failure;
            stop
        endif

    split again
        :Cancel order;
    end split
fork again
    :View profile;
    split
        :Change profile;
    split again
        :Change password;
    end split
fork again
    :View histories;
end fork

#CCFFFF:Sign out;

end
